ðŸ”¸ Problem 1: Count Character Frequency

Given a string, return an object that shows how many times each letter appears. Ignore punctuation and spaces, and treat uppercase and lowercase as the same.

Example Input:

const input = "Hello, World!";

ðŸ”¸ Your Task to Perform

Loop through each character, convert to lowercase, and track frequency using an object.

Sample Code:

const freq = {};
for (let ch of input.toLowerCase()) {
  if (ch >= 'a' && ch <= 'z') {
    freq[ch] = (freq[ch] || 0) + 1;
  }
}
console.log(freq);

Example Output: 

{
  h: 1, e: 1, l: 3,
  o: 2, w: 1, r: 1,
  d: 1
}

ðŸ”¸ Problem 2: Reverse a String

Return the reverse of a given string. No built-in reverse functions allowed â€” try doing it manually.

Sample Code:

let str = "openai";
let reversed = "";
for (let i = str.length - 1; i >= 0; i--) {
  reversed += str[i];
}
console.log(reversed); // "ianepo"

ðŸ”¸ Problem 3: Check for Palindrome

A palindrome reads the same backward. Write a function to check if a string is a palindrome or not. 

function isPalindrome(str) {
  let clean = str.toLowerCase().replace(/[^a-z0-9]/g, "");
  let reversed = clean.split("").reverse().join("");
  return clean === reversed;
}

console.log(isPalindrome("Racecar")); // true
console.log(isPalindrome("hello"));   // false

Bonus Problem: Capitalize First Letter of Each Word

Input: "hello world" â†’ Output: "Hello World" 

function capitalizeWords(str) {
  return str
    .split(" ")
    .map(word => word.charAt(0).toUpperCase() + word.slice(1))
    .join(" ");
}

import pygame
import sys
import random
import math

# Initialize pygame
pygame.init()

# Screen dimensions
WIDTH, HEIGHT = 400, 600
screen = pygame.display.set_mode((WIDTH, HEIGHT))
pygame.display.set_caption("Flappy Bird Deluxe")

# Colors
SKY_BLUE = (135, 206, 235)
GREEN = (0, 180, 40)
DARK_GREEN = (0, 160, 30)
WHITE = (255, 255, 255)
BLACK = (0, 0, 0)
YELLOW = (255, 215, 0)
ORANGE = (255, 165, 0)
RED = (255, 50, 50)
PIPE_GREEN = (30, 180, 70)
BG_HILLS = (100, 200, 100)

# Game states
MENU = 0
PLAYING = 1
GAME_OVER = 2
PAUSED = 3

# Game variables
gravity = 0.5
flap_strength = -9
pipe_speed = 3
pipe_frequency = 1800  # milliseconds
ground_height = 100
score = 0
high_score = 0
game_state = MENU

# Clock for controlling frame rate
clock = pygame.time.Clock()
FPS = 60

# Fonts
font_small = pygame.font.SysFont("Arial", 24)
font_medium = pygame.font.SysFont("Arial", 36)
font_large = pygame.font.SysFont("Arial", 72)

# Bird class with advanced physics
class Bird:
    def __init__(self):
        self.x = WIDTH // 3
        self.y = HEIGHT // 2
        self.radius = 15
        self.velocity = 0
        self.angle = 0
        self.flap_time = 0
        self.wing_up = False
        self.color = YELLOW
        self.trail = []
        self.trail_timer = 0
        
    def flap(self):
        self.velocity = flap_strength
        self.flap_time = 5
        self.wing_up = True
        
    def update(self):
        # Apply gravity with air resistance
        air_resistance = 0.98
        self.velocity += gravity
        self.velocity *= air_resistance
        
        # Update position
        self.y += self.velocity
        
        # Rotate bird based on velocity
        self.angle = -self.velocity * 2
        self.angle = max(-30, min(self.angle, 90))
        
        # Animate wings
        if self.flap_time > 0:
            self.flap_time -= 1
        else:
            self.wing_up = not self.wing_up
            
        # Add trail effect
        self.trail_timer += 1
        if self.trail_timer > 2:
            self.trail.append((self.x, self.y))
            self.trail_timer = 0
            
        # Limit trail length
        if len(self.trail) > 10:
            self.trail.pop(0)
            
    def draw(self):
        # Draw trail
        for i, (trail_x, trail_y) in enumerate(self.trail):
            alpha = i / len(self.trail) * 100
            trail_color = (self.color[0], self.color[1], self.color[2], int(alpha))
            trail_surf = pygame.Surface((self.radius*2, self.radius*2), pygame.SRCALPHA)
            pygame.draw.circle(trail_surf, trail_color, (self.radius, self.radius), self.radius * (i+1)/len(self.trail))
            screen.blit(trail_surf, (trail_x - self.radius, trail_y - self.radius))
        
        # Draw bird body
        rotated_bird = pygame.Surface((self.radius*2, self.radius*2), pygame.SRCALPHA)
        pygame.draw.circle(rotated_bird, self.color, (self.radius, self.radius), self.radius)
        
        # Draw eye
        pygame.draw.circle(rotated_bird, WHITE, (self.radius + 5, self.radius - 5), 5)
        pygame.draw.circle(rotated_bird, BLACK, (self.radius + 6, self.radius - 4), 2)
        
        # Draw beak
        pygame.draw.polygon(rotated_bird, ORANGE, [(self.radius + 10, self.radius), 
                                                 (self.radius + 20, self.radius - 3),
                                                 (self.radius + 20, self.radius + 3)])
        
        # Draw wing
        wing_y = self.radius + (5 if self.wing_up else -5)
        pygame.draw.ellipse(rotated_bird, ORANGE, (self.radius - 15, wing_y, 15, 10))
        
        # Rotate the bird
        rotated_bird = pygame.transform.rotate(rotated_bird, self.angle)
        screen.blit(rotated_bird, (self.x - rotated_bird.get_width()//2, self.y - rotated_bird.get_height()//2))
        
    def get_mask(self):
        return pygame.Rect(self.x - self.radius, self.y - self.radius, self.radius*2, self.radius*2)

# Pipe class
class Pipe:
    def __init__(self):
        self.x = WIDTH + 100
        self.gap_y = random.randint(150, HEIGHT - 150 - ground_height)
        self.gap_size = 150
        self.width = 70
        self.passed = False
        self.color = PIPE_GREEN
        
    def update(self):
        self.x -= pipe_speed
        
    def draw(self):
        # Draw top pipe
        pygame.draw.rect(screen, self.color, (self.x, 0, self.width, self.gap_y - self.gap_size//2))
        pygame.draw.rect(screen, DARK_GREEN, (self.x - 5, self.gap_y - self.gap_size//2 - 20, self.width + 10, 20))
        
        # Draw bottom pipe
        bottom_pipe_y = self.gap_y + self.gap_size//2
        pygame.draw.rect(screen, self.color, (self.x, bottom_pipe_y, self.width, HEIGHT - bottom_pipe_y - ground_height))
        pygame.draw.rect(screen, DARK_GREEN, (self.x - 5, bottom_pipe_y, self.width + 10, 20))
        
    def collide(self, bird):
        bird_rect = bird.get_mask()
        
        # Top pipe collision
        if bird_rect.colliderect(pygame.Rect(self.x, 0, self.width, self.gap_y - self.gap_size//2)):
            return True
            
        # Bottom pipe collision
        if bird_rect.colliderect(pygame.Rect(self.x, self.gap_y + self.gap_size//2, self.width, HEIGHT)):
            return True
            
        return False

# Cloud class for background
class Cloud:
    def __init__(self):
        self.x = WIDTH + random.randint(0, 100)
        self.y = random.randint(50, HEIGHT//2)
        self.speed = random.uniform(0.5, 1.5)
        self.size = random.randint(30, 70)
        
    def update(self):
        self.x -= self.speed
        if self.x < -100:
            self.x = WIDTH + random.randint(0, 100)
            self.y = random.randint(50, HEIGHT//2)
            
    def draw(self):
        pygame.draw.circle(screen, (240, 240, 240), (int(self.x), int(self.y)), self.size)
        pygame.draw.circle(screen, (240, 240, 240), (int(self.x) + self.size//2, int(self.y) - self.size//3), self.size//1.5)
        pygame.draw.circle(screen, (240, 240, 240), (int(self.x) - self.size//2, int(self.y) - self.size//3), self.size//1.5)

# Button class for menus
class Button:
    def __init__(self, x, y, width, height, text, color, hover_color):
        self.rect = pygame.Rect(x, y, width, height)
        self.text = text
        self.color = color
        self.hover_color = hover_color
        self.is_hovered = False
        
    def draw(self):
        color = self.hover_color if self.is_hovered else self.color
        pygame.draw.rect(screen, color, self.rect, border_radius=10)
        pygame.draw.rect(screen, BLACK, self.rect, 3, border_radius=10)
        
        text_surface = font_medium.render(self.text, True, WHITE)
        text_rect = text_surface.get_rect(center=self.rect.center)
        screen.blit(text_surface, text_rect)
        
    def check_hover(self, pos):
        self.is_hovered = self.rect.collidepoint(pos)
        return self.is_hovered
        
    def check_click(self, pos, event):
        if event.type == pygame.MOUSEBUTTONDOWN and event.button == 1:
            return self.rect.collidepoint(pos)
        return False

# Create game objects
bird = Bird()
pipes = []
clouds = [Cloud() for _ in range(5)]
last_pipe = pygame.time.get_ticks()

# Create buttons
play_button = Button(WIDTH//2 - 100, HEIGHT//2, 200, 50, "Play", GREEN, DARK_GREEN)
restart_button = Button(WIDTH//2 - 100, HEIGHT//2 + 60, 200, 50, "Restart", GREEN, DARK_GREEN)
menu_button = Button(WIDTH//2 - 100, HEIGHT//2 + 120, 200, 50, "Menu", (100, 100, 240), (80, 80, 220))
resume_button = Button(WIDTH//2 - 100, HEIGHT//2 - 60, 200, 50, "Resume", GREEN, DARK_GREEN)

# Draw background with hills
def draw_background():
    # Sky gradient
    for y in range(HEIGHT):
        color_factor = 1 - (y / HEIGHT) * 0.5
        color = (int(135 * color_factor), int(206 * color_factor), int(235 * color_factor))
        pygame.draw.line(screen, color, (0, y), (WIDTH, y))
    
    # Draw hills
    for i in range(5):
        hill_y = HEIGHT - ground_height - 50 - i*30
        hill_width = WIDTH // (i+1)
        for x in range(0, WIDTH, 2):
            hill_height = math.sin(x/hill_width * math.pi) * 20
            pygame.draw.line(screen, BG_HILLS, (x, hill_y + hill_height), (x, HEIGHT))
    
    # Draw clouds
    for cloud in clouds:
        cloud.update()
        cloud.draw()
    
    # Draw ground
    pygame.draw.rect(screen, GREEN, (0, HEIGHT - ground_height, WIDTH, ground_height))
    
    # Draw ground details
    for i in range(20):
        x = random.randint(0, WIDTH)
        y = HEIGHT - random.randint(10, ground_height)
        pygame.draw.circle(screen, DARK_GREEN, (x, y), 3)

# Draw menu screen
def draw_menu():
    draw_background()
    
    # Draw title
    title_text = font_large.render("Flappy Bird", True, WHITE)
    screen.blit(title_text, (WIDTH//2 - title_text.get_width()//2, 100))
    
    # Draw subtitle
    subtitle_text = font_medium.render("Deluxe Edition", True, YELLOW)
    screen.blit(subtitle_text, (WIDTH//2 - subtitle_text.get_width()//2, 180))
    
    # Draw buttons
    play_button.draw()
    
    # Draw high score
    if high_score > 0:
        score_text = font_small.render(f"High Score: {high_score}", True, WHITE)
        screen.blit(score_text, (WIDTH//2 - score_text.get_width()//2, HEIGHT - 100))
    
    # Draw instructions
    instructions = font_small.render("Press SPACE or CLICK to flap", True, WHITE)
    screen.blit(instructions, (WIDTH//2 - instructions.get_width()//2, HEIGHT - 150))

# Draw game over screen
def draw_game_over():
    # Semi-transparent overlay
    overlay = pygame.Surface((WIDTH, HEIGHT), pygame.SRCALPHA)
    overlay.fill((0, 0, 0, 150))
    screen.blit(overlay, (0, 0))
    
    # Draw game over text
    game_over_text = font_large.render("Game Over", True, RED)
    screen.blit(game_over_text, (WIDTH//2 - game_over_text.get_width()//2, 150))
    
    # Draw score
    score_text = font_medium.render(f"Score: {score}", True, WHITE)
    screen.blit(score_text, (WIDTH//2 - score_text.get_width()//2, 250))
    
    # Draw high score
    high_score_text = font_medium.render(f"High Score: {high_score}", True, YELLOW)
    screen.blit(high_score_text, (WIDTH//2 - high_score_text.get_width()//2, 300))
    
    # Draw buttons
    restart_button.draw()
    menu_button.draw()

# Draw paused screen
def draw_paused():
    # Semi-transparent overlay
    overlay = pygame.Surface((WIDTH, HEIGHT), pygame.SRCALPHA)
    overlay.fill((0, 0, 0, 150))
    screen.blit(overlay, (0, 0))
    
    # Draw paused text
    paused_text = font_large.render("Paused", True, WHITE)
    screen.blit(paused_text, (WIDTH//2 - paused_text.get_width()//2, 150))
    
    # Draw buttons
    resume_button.draw()
    menu_button.draw()

# Draw HUD during gameplay
def draw_hud():
    # Draw score
    score_text = font_medium.render(str(score), True, WHITE)
    screen.blit(score_text, (WIDTH//2 - score_text.get_width()//2, 50))
    
    # Draw pause button
    pygame.draw.rect(screen, (100, 100, 100, 180), (WIDTH - 50, 20, 30, 30), border_radius=5)
    pygame.draw.line(screen, WHITE, (WIDTH - 40, 25), (WIDTH - 40, 45), 3)
    pygame.draw.line(screen, WHITE, (WIDTH - 30, 25), (WIDTH - 30, 45), 3)

# Main game loop
running = True
while running:
    # Handle events
    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            running = False
            
        if event.type == pygame.KEYDOWN:
            if event.key == pygame.K_ESCAPE:
                if game_state == PLAYING:
                    game_state = PAUSED
                elif game_state == PAUSED:
                    game_state = PLAYING
                    
            if event.key == pygame.K_SPACE:
                if game_state == PLAYING:
                    bird.flap()
                elif game_state == MENU:
                    game_state = PLAYING
                    score = 0
                    bird = Bird()
                    pipes = []
                    
        if event.type == pygame.MOUSEBUTTONDOWN:
            if game_state == PLAYING:
                bird.flap()
                
            # Check for button clicks
            mouse_pos = pygame.mouse.get_pos()
            
            if game_state == MENU:
                if play_button.check_click(mouse_pos, event):
                    game_state = PLAYING
                    score = 0
                    bird = Bird()
                    pipes = []
                    
            elif game_state == GAME_OVER:
                if restart_button.check_click(mouse_pos, event):
                    game_state = PLAYING
                    score = 0
                    bird = Bird()
                    pipes = []
                elif menu_button.check_click(mouse_pos, event):
                    game_state = MENU
                    
            elif game_state == PAUSED:
                if resume_button.check_click(mouse_pos, event):
                    game_state = PLAYING
                elif menu_button.check_click(mouse_pos, event):
                    game_state = MENU
                    
            # Check for pause button click
            if game_state == PLAYING and mouse_pos[0] > WIDTH - 50 and mouse_pos[0] < WIDTH - 20 and mouse_pos[1] > 20 and mouse_pos[1] < 50:
                game_state = PAUSED
    
    # Get mouse position for button hover effects
    mouse_pos = pygame.mouse.get_pos()
    if game_state == MENU:
        play_button.check_hover(mouse_pos)
    elif game_state == GAME_OVER:
        restart_button.check_hover(mouse_pos)
        menu_button.check_hover(mouse_pos)
    elif game_state == PAUSED:
        resume_button.check_hover(mouse_pos)
        menu_button.check_hover(mouse_pos)
    
    # Draw background
    draw_background()
    
    if game_state == MENU:
        # Draw menu
        draw_menu()
        
        # Animate bird on menu screen
        bird.velocity = math.sin(pygame.time.get_ticks() / 200) * 2
        bird.update()
        bird.draw()
        
    elif game_state == PLAYING:
        # Update bird
        bird.update()
        
        # Generate pipes
        time_now = pygame.time.get_ticks()
        if time_now - last_pipe > pipe_frequency:
            pipes.append(Pipe())
            last_pipe = time_now
            
        # Update and draw pipes
        pipe_removal = []
        for pipe in pipes:
            pipe.update()
            
            # Check if pipe is off screen
            if pipe.x < -pipe.width:
                pipe_removal.append(pipe)
                
            # Check if bird passed pipe
            if not pipe.passed and pipe.x < bird.x:
                pipe.passed = True
                score += 1
                
            # Check for collision
            if pipe.collide(bird):
                game_state = GAME_OVER
                if score > high_score:
                    high_score = score
                
            pipe.draw()
            
        # Remove off-screen pipes
        for pipe in pipe_removal:
            pipes.remove(pipe)
            
        # Check for collision with ground or ceiling
        if bird.y > HEIGHT - ground_height - bird.radius or bird.y < bird.radius:
            game_state = GAME_OVER
            if score > high_score:
                high_score = score
                
        # Draw bird
        bird.draw()
        
        # Draw HUD
        draw_hud()
        
    elif game_state == GAME_OVER:
        # Draw pipes
        for pipe in pipes:
            pipe.draw()
            
        # Draw bird
        bird.draw()
        
        # Draw game over screen
        draw_game_over()
        
    elif game_state == PAUSED:
        # Draw pipes
        for pipe in pipes:
            pipe.draw()
            
        # Draw bird
        bird.draw()
        
        # Draw HUD
        draw_hud()
        
        # Draw paused screen
        draw_paused()
    
    # Update display
    pygame.display.flip()
    
    # Control frame rate
    clock.tick(FPS)

# Quit pygame
pygame.quit()
sys.exit()
